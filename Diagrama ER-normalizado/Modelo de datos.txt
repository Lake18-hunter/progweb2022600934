Modelo de datos

Marcas (
    id_marca INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(150) NOT NULL,
    url_marca VARCHAR(255)
);

Categorias (
    id_categoria INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(150) NOT NULL,
    url_categoria VARCHAR(255),
    id_padre INT,
    FOREIGN KEY (id_padre) REFERENCES Categorias(id_categoria)
);

Productos (
    sku VARCHAR(50) PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    descripcion TEXT,
    precio DECIMAL(10,2) NOT NULL,
    id_marca INT NULL,  -- 
    FOREIGN KEY (id_marca) REFERENCES Marcas(id_marca)
);

Multimedia (
    id_multimedia INT AUTO_INCREMENT PRIMARY KEY,
    url_recurso VARCHAR(255) NOT NULL,
    tipo VARCHAR(50) NOT NULL,
    sku VARCHAR(50) NOT NULL,
    FOREIGN KEY (sku) REFERENCES Productos(sku)
);

Producto_Categoria (
    sku VARCHAR(50) NOT NULL,
    id_categoria INT NOT NULL,
    PRIMARY KEY (sku, id_categoria),
    FOREIGN KEY (sku) REFERENCES Productos(sku),
    FOREIGN KEY (id_categoria) REFERENCES Categorias(id_categoria)
);
